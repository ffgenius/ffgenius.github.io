Echarts.vue

vue
<template>
  <div :id="nanoid" :style="{width, height}"></div>
</template>

<script>
import { nanoid } from 'nanoid'
import * as echarts from 'echarts'
import { debounce } from 'lodash'

export default {
  name: 'Echarts',
  props: {
    width: {
      type: String,
      default: '100%'
    },
    height: {
      type: String,
      default: '100%'
    },
    options: {
      type: Object
    }
  },
  data() {
    return {
      nanoid: nanoid(),
      chart: null
    }
  },
  mounted() {
    this.$nextTick(() => {
      this.initCharts()
    })
  },
  watch: {
    options: {
      deep: true,
      handler() {
        this.$nextTick(() => {
          this.initCharts()
        })
      }
    }
  },
  methods: {
    initCharts() {
      if (!document.getElementById(this.nanoid)) return
      this.chart = echarts.init(document.getElementById(this.nanoid))
      this.chart.setOption(this.options)
      // 引入debounce，防止频繁更改浏览页尺寸出现页面抖动
      window.addEventListener('resize', debounce(() => {
        if (this.chart) {
          this.chart.resize()
        }
      }, 100))
    }
  },
  // 组件销毁 echarts 销毁
  beforeDestroy() {
    if (!this.chart) {
      return;
    }
    this.chart.dispose();
    this.chart = null;
  }
}
</script>
